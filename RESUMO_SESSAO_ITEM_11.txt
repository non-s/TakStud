PADRÃO 4: FILTERED QUERY COM EMPTY CHECK
Antes (42 linhas):
    fun getStudentsForTeacher(teacherId: String, classNames: List<String>): Flow<List<Student>> = callbackFlow {
        if (classNames.isEmpty()) {
            trySend(emptyList())
            close()
            return@callbackFlow
        }
        val listener = Firebase.firestore.collection("students")
            .whereIn("studentClass", classNames)
            .addSnapshotListener { snapshots, e ->
            if (e != null) {
                Log.e(TAG, "Erro ao buscar students do teacher", e)
                close(e)
                return@addSnapshotListener
            }
            val students = snapshots?.mapNotNull { doc ->
                try {
                    doc.toObject(Student::class.java).copy(id = doc.id)
                } catch (ex: Exception) {
                    Log.e(TAG, "Erro ao converter student", ex)
                    null
                }
            } ?: emptyList()
            Log.i(TAG, "Students carregados para teacher $teacherId: ${students.size}")
            trySend(students)
        }
        awaitClose { listener.remove() }
    }

Depois (12 linhas):
    fun getStudentsForTeacher(teacherId: String, classNames: List<String>): Flow<List<Student>> {
        Log.i(TAG, "Buscando students para teacher $teacherId em ${classNames.size} turmas")
        return if (classNames.isEmpty()) {
            flowOf(emptyList())
        } else {
            firestoreQueryFlow(
                Firebase.firestore.collection("students")
                    .whereIn("studentClass", classNames),
                Student::class.java,
                TAG
            )
        }
    }
